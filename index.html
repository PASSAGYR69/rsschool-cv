//Example: n = 86240 should return "(2**5)(5)(7**2)(11)"

function primeFactors(n) { 
  let arr = []
  let finalStr = ''

  prime:
  for (i = 2; i < Math.sqrt(n); i++ ) {
    for (j = 2; j < i; j++) {
      if (i % j == 0) continue prime;
    }
    while(true) {
      if (n % i == 0) {n = n / i; arr.push (i)}
      if (n % i != 0) {break}
    }
  }
  if(n > 1) arr.push(n);
  let step = 1;

  for (i = 0; i < arr.length; i++) {
    if (arr[i] == arr [i+1]) {
      step += 1
    }
    if (arr[i] != arr [i+1]) {
      step == 1 ?  finalStr += '(' + arr[i]  + ')' :
      finalStr += '(' + arr[i] +'**' + step + ')' ;
      step = 1;
    }
  }

  return (finalStr)
}
